üìÑ docker-compose.yml

  1. Install Docker
  2. Create a file named docker-compose.yml in the project root directory. Inside it, copy the codes below and paste: 

version: '3.9'
services:
  mysql:
    image: mysql:latest
    restart: always
    environment:
      MYSQL_ROOT_PASSWORD: "yagya"
      MYSQL_DATABASE: your_database_name
      MYSQL_USER: yagya
      MYSQL_PASSWORD: yagya
    ports:
      - "3310:3306"
    volumes:
      - ./.srv/database:/var/lib/mysql

  wordpress:
    image: wordpress:latest
    restart: always
    environment:
      WORDPRESS_DB_HOST: mysql:3306
      WORDPRESS_DB_USER: yagya
      WORDPRESS_DB_PASSWORD: yagya
      WORDPRESS_DB_NAME: your_database_name
    ports:
      - "8080:80"
    depends_on:
      - mysql
    volumes:
      - ./.srv/wordpress:/var/www/html
      - ./themes/:/var/www/html/wp-content/themes
      - ./plugins/:/var/www/html/wp-content/plugins
      - ./mu-plugins/:/var/www/html/wp-content/mu-plugins
      - ./.srv/custom.ini:/usr/local/etc/php/conf.d/custom.ini
    links:
      - mysql
    
  phpmyadmin:
    image: phpmyadmin/phpmyadmin:latest
    environment:
      MYSQL_USERNAME: yagya
      MYSQL_ROOT_PASSWORD: yagya
      PMA_HOST: mysql
    ports:
      - "8181:80"
    depends_on:
      - mysql
    links:
      - mysql

‚ö° Docker Commands

Runs all services defined in docker-compose.yml in detached mode.

docker-compose up -d

Stops running containers and removes them along with networks created.

docker-compose down

Follows and displays logs of all running containers.

docker-compose logs -f

Restarts all running containers defined in docker-compose.yml.

docker-compose restart
